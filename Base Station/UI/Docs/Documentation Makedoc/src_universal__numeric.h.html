#ifndef __UNIVERSALNUMERIC_H__
#define __UNIVERSALNUMERIC_H__

#include "universal.h"
#include "universal__String.h"

namespace universal {
  
  /** 
   * Built-in token base classes.
   */

  class Float;
  class Double;
  class UnsignedInteger;

  /**
   * secondary:
   * This is a very simple component whose sole purpose is to wrap a
   * signed integer value. The precise type is "long". 
   *
   * The cast operator to long() is often useful for getting the 
   * value of a long integer token as a long. 
   * This wrapper class features =,+,-,/ and * operators taking all the 
   * other possible numeric types.
   *
   * It also features a conversion function to a universal::String.
   */

  class UNIVERSAL_DLL SignedInteger : public Object
    {
    private:      
      long m_lValue;
    public:
      SignedInteger();
      virtual ~SignedInteger();
      operator long() const;
      SignedInteger& operator =(long);
      SignedInteger& operator =(short);
      SignedInteger& operator =(int);
      SignedInteger& operator =(unsigned long);
      SignedInteger& operator =(unsigned short);
      SignedInteger& operator =(unsigned int);
      SignedInteger& operator =(float);
      SignedInteger& operator =(double);
      String toString() const;

      bool even() const;
      bool odd() const;

      float operator+(const Float& f);
      double operator+(const Double& d);
      long operator+(const SignedInteger& si);
      long operator+(const UnsignedInteger& ui);
      float operator-(const Float& f);
      double operator-(const Double& d);
      long operator-(const SignedInteger& si);
      long operator-(const UnsignedInteger& ui);
      float operator*(const Float& f);
      double operator*(const Double& d);
      long operator*(const SignedInteger& si);
      long operator*(const UnsignedInteger& ui);
      float operator/(const Float& f);
      double operator/(const Double& d);
      long operator/(const SignedInteger& si);
      long operator/(const UnsignedInteger& ui);
    };
